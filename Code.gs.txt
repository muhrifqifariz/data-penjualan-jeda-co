/**
 * KODE INI DITEMPEL DI GOOGLE APPS SCRIPT.
 * Cara pakai lengkap ada di PANDUAN_SETUP.txt (untuk Jeda Rider) dan
 * jeda-admin/PANDUAN_JEDA_ADMIN.txt (untuk Jeda Admin).
 *
 * ISI FILE INI:
 * 1. setupSheets()  -> jalankan SEKALI (aman dijalankan ulang) untuk membuat
 *                      sheet Menu, Rider, Aturan, Log, Setoran, Rekap, dan
 *                      MENGUPGRADE struktur lama ke versi baru tanpa
 *                      menghapus data (sub-kategori minuman, konversi cup,
 *                      ID Submission untuk revisi lewat Jeda Admin).
 * 2. doGet()        -> dipanggil Jeda Rider untuk ambil menu/rider/aturan,
 *                      dan mode real-time (?action=getSession) untuk
 *                      memuat ulang tally yang sudah tersimpan hari itu;
 *                      dipanggil Jeda Admin (pakai ?action=listSubmissions
 *                      atau ?action=getSubmission&id=...) untuk lihat &
 *                      buka detail submission yang mau direvisi.
 * 3. doPost()       -> dipanggil Jeda Rider tiap tap +/- (adjustItem) atau
 *                      isi angka langsung (setItemField) -> real-time
 *                      tersimpan ke sheet Log per rider+tanggal+menu;
 *                      closeSession dipanggil saat "Tutup Hari Ini" utk
 *                      simpan ringkasan Modal/QRIS/Komisi/Setoran;
 *                      juga dipanggil utk kelola menu (addMenu/dst), dan
 *                      dipanggil Jeda Admin (verifyPin/updateSubmission/
 *                      deleteSubmission) untuk revisi data, dilindungi PIN.
 *
 * Catatan:
 * - Tarif & batas komisi ada di sheet "Aturan" — bisa diubah langsung di
 *   situ kapan saja, TIDAK perlu edit kode ini.
 * - Daftar nama rider ada di sheet "Rider" — tambahkan nama baru di
 *   kolom A, satu nama per baris, TIDAK perlu edit kode ini.
 * - PIN untuk Jeda Admin ada di sheet "Aturan" sel B9, ganti kapan saja.
 * - Sub-kategori Minuman (Cup / 500 ml / 1 Liter) & faktor konversi cup-nya
 *   (1 / 4 / 8) diatur di dalam kode ini lewat deriveSatuanKonversi di bawah.
 */

const SHEET_MENU    = "Menu";
const SHEET_LOG     = "Log";
const SHEET_REKAP   = "Rekap";
const SHEET_ATURAN  = "Aturan";
const SHEET_RIDER   = "Rider";
const SHEET_SETORAN = "Setoran";
const JUMLAH_SLOT_MENU = 30; // kapasitas baris menu di Rekap, boleh diperbesar

// Sub-kategori Minuman & faktor konversi ke "cup" untuk keperluan komisi.
// 1 botol 500ml = 4 cup, 1 botol 1 Liter = 8 cup (sesuai aturan bisnis).
// Dipakai untuk MENGHITUNG nilai Satuan & Konversi Cup secara langsung
// (ditulis sebagai nilai biasa ke sheet, BUKAN rumus) supaya selalu akurat
// untuk menu baru yang ditambah lewat aplikasi maupun manual di sheet.
function deriveSatuanKonversi(kategori, subKategori) {
  if (kategori === "Cookies") return { satuan: "pcs", konversi: 1 };
  if (subKategori === "500 ml") return { satuan: "botol", konversi: 4 };
  if (subKategori === "1 Liter") return { satuan: "botol", konversi: 8 };
  return { satuan: "cup", konversi: 1 }; // default: Cup
}

// ---------------------------------------------------------------------
// Cari baris terakhir yang BENAR-BENAR berisi data di kolom tertentu,
// dengan cara scan langsung (tidak pakai sheet.getLastRow() yang bisa
// salah baca kalau ada sisa formula/nilai kosong "" di baris bawahnya).
// col: nomor kolom (1 = A, 2 = B, dst). Return 1 kalau tidak ada data
// sama sekali (hanya header).
// ---------------------------------------------------------------------
function findLastDataRow(sheet, col) {
  const maxRows = sheet.getMaxRows();
  const values = sheet.getRange(1, col, maxRows, 1).getValues();
  for (let i = values.length - 1; i >= 1; i--) {
    if (values[i][0] !== "" && values[i][0] !== null) return i + 1;
  }
  return 1;
}

// Bersihkan total sisa data/format kosong di bawah baris data asli,
// supaya findLastDataRow/getLastRow ke depannya selalu akurat.
function clearGarbageBelow(sheet, lastRealRow, numCols) {
  const maxRows = sheet.getMaxRows();
  if (maxRows > lastRealRow) {
    sheet.getRange(lastRealRow + 1, 1, maxRows - lastRealRow, numCols).clearContent();
  }
}

// =====================================================================
// 1. SETUP OTOMATIS — jalankan sekali (atau ulang) lewat editor Apps Script:
//    pilih fungsi "setupSheets" di dropdown atas, lalu klik Run.
// =====================================================================
function setupSheets() {
  const ss = SpreadsheetApp.getActiveSpreadsheet();

  const menuSheet = setupMenuSheet(ss);
  setupAturanSheet(ss);
  const riderSheet = setupRiderSheet(ss);
  setupLogSheet(ss);
  setupSetoranSheet(ss);
  setupRekapSheet(ss, menuSheet, riderSheet);

  SpreadsheetApp.getUi().alert("Setup selesai! Sheet Menu, Rider, Aturan, Log, Setoran, dan Rekap sudah siap.");
}

// ---------------------------------------------------------------------
// Sheet Menu — kolom: Menu | Kategori | Harga | Sub Kategori | Satuan | Konversi Cup
// Dibuat kalau belum ada. Kalau sudah ada tapi masih format lama (tanpa
// kolom Sub Kategori), otomatis di-UPGRADE tanpa menghapus data yang ada.
// Kolom Satuan & Konversi Cup diisi NILAI LANGSUNG (bukan rumus) supaya
// tidak ada risiko salah/kosong untuk baris yang ditambah belakangan.
// Sisa data kosong dari versi lama (kalau ada) DIBERSIHKAN TOTAL supaya
// baris terakhir selalu terbaca akurat.
// ---------------------------------------------------------------------
function setupMenuSheet(ss) {
  let sheet = ss.getSheetByName(SHEET_MENU);

  if (!sheet) {
    sheet = ss.insertSheet(SHEET_MENU);
    sheet.getRange(1, 1, 1, 6).setValues([[
      "Menu", "Kategori", "Harga", "Sub Kategori", "Satuan", "Konversi Cup"
    ]]).setFontWeight("bold").setBackground("#1F6F50").setFontColor("#FFFFFF");
    const seed = [
      ["Coco Latte", "Minuman", 12000, "Cup"],
      ["Coco Bold", "Minuman", 15000, "Cup"],
      ["Coco Matcha", "Minuman", 15000, "Cup"],
      ["Coco Choco", "Minuman", 13000, "Cup"],
      ["Coco Americano", "Minuman", 8000, "Cup"],
      ["Soft Cookies", "Cookies", 8000, ""]
    ].map(row => {
      const sk = deriveSatuanKonversi(row[1], row[3]);
      return [row[0], row[1], row[2], row[3], sk.satuan, sk.konversi];
    });
    sheet.getRange(2, 1, seed.length, 6).setValues(seed);
    sheet.setFrozenRows(1);
    sheet.autoResizeColumns(1, 6);
    return sheet;
  }

  // ---- Upgrade dari format lama (3 kolom) ke format baru (6 kolom) ----
  const header = sheet.getRange(1, 1, 1, Math.max(sheet.getLastColumn(), 6)).getValues()[0];
  if (header[3] !== "Sub Kategori") {
    const lastRealRow = findLastDataRow(sheet, 1); // scan kolom A (Menu)
    const oldData = lastRealRow > 1 ? sheet.getRange(2, 1, lastRealRow - 1, 3).getValues() : [];

    sheet.getRange(1, 1, 1, 6).setValues([[
      "Menu", "Kategori", "Harga", "Sub Kategori", "Satuan", "Konversi Cup"
    ]]).setFontWeight("bold").setBackground("#1F6F50").setFontColor("#FFFFFF");

    if (oldData.length) {
      const migrated = oldData.filter(row => row[0]).map(row => {
        const kategori = row[1];
        const subKategori = kategori === "Cookies" ? "" : "Cup"; // default aman: minuman lama = Cup
        const sk = deriveSatuanKonversi(kategori, subKategori);
        return [row[0], kategori, row[2], subKategori, sk.satuan, sk.konversi];
      });
      if (migrated.length) sheet.getRange(2, 1, migrated.length, 6).setValues(migrated);
      clearGarbageBelow(sheet, migrated.length + 1, 6);
    } else {
      clearGarbageBelow(sheet, 1, 6);
    }
    sheet.autoResizeColumns(1, 6);
  } else {
    // ---- Sudah format baru: perbaiki ulang Satuan/Konversi Cup untuk
    // SEMUA baris nyata (pakai scan kolom A, bukan getLastRow()), lalu
    // BERSIHKAN TOTAL sisa data/formula kosong di bawahnya. Ini juga
    // memperbaiki bug lama di mana getLastRow() salah baca sampai baris
    // 1000-an gara-gara sisa ARRAYFORMULA, yang bikin menu baru "hilang"
    // (tersimpan tapi di baris sangat jauh) atau gagal ditambah.
    const lastRealRow = findLastDataRow(sheet, 1);
    if (lastRealRow > 1) {
      const data = sheet.getRange(2, 1, lastRealRow - 1, 4).getValues();
      const fixed = data.map(row => {
        if (!row[0]) return ["", ""];
        const sk = deriveSatuanKonversi(row[1], row[3]);
        return [sk.satuan, sk.konversi];
      });
      sheet.getRange(2, 5, fixed.length, 2).setValues(fixed);
    }
    clearGarbageBelow(sheet, lastRealRow, 6);
  }
  return sheet;
}

// Sheet Aturan Komisi — dibuat HANYA jika belum ada.
// ---------------------------------------------------------------------
function setupAturanSheet(ss) {
  let sheet = ss.getSheetByName(SHEET_ATURAN);

  if (sheet) {
    // Sudah ada (user lama) — cek apakah baris Operasional sudah ada,
    // kalau belum, TAMBAHKAN tanpa mengubah tarif Minuman/Cookies yang ada.
    const labelA5 = sheet.getRange("A5").getValue();
    if (labelA5 !== "Biaya Operasional (Rp)") {
      sheet.getRange("A5").setValue("Biaya Operasional (Rp)").setFontWeight("bold");
      sheet.getRange("B5").setValue(10000).setNumberFormat("Rp#,##0");
    }
    const labelA9 = sheet.getRange("A9").getValue();
    if (labelA9 !== "PIN Admin (Jeda Admin)") {
      sheet.getRange("A9").setValue("PIN Admin (Jeda Admin)").setFontWeight("bold");
      sheet.getRange("B9").setValue("1234").setNumberFormat("@");
    }
    return sheet;
  }

  sheet = ss.insertSheet(SHEET_ATURAN);
  sheet.getRange(1, 1, 1, 4).setValues([[
    "Kategori", "Batas Tier 1", "Tarif Tier 1 (Rp)", "Tarif Tier 2 (Rp)"
  ]]).setFontWeight("bold").setBackground("#1F6F50").setFontColor("#FFFFFF");
  sheet.getRange(2, 1, 2, 4).setValues([
    ["Minuman", 50, 1000, 2000],
    ["Cookies", 50, 500, 1000]
  ]);
  sheet.getRange("C2:D3").setNumberFormat("Rp#,##0");
  sheet.setFrozenRows(1);

  sheet.getRange("A5").setValue("Biaya Operasional (Rp)").setFontWeight("bold");
  sheet.getRange("B5").setValue(10000).setNumberFormat("Rp#,##0");

  sheet.getRange("A9").setValue("PIN Admin (Jeda Admin)").setFontWeight("bold");
  sheet.getRange("B9").setValue("1234").setNumberFormat("@");

  sheet.autoResizeColumns(1, 4);
  sheet.getRange("A7").setValue(
    "Catatan: ubah angka di baris Minuman/Cookies di atas kapan saja. " +
    "Untuk Minuman, satuannya cup (termasuk cup setara dari botol 500ml/1L). " +
    "Untuk Cookies, satuannya pcs. Biaya Operasional di B5 dipakai untuk " +
    "menghitung Setoran Cash di aplikasi, ubah kapan saja kalau perlu. " +

    "kapan saja demi keamanan. Semua ini otomatis diikuti, tidak perlu ubah kode.")

// ---------------------------------------------------------------------
// Sheet Log — kolom A-J sama seperti sebelumnya (kompatibel dengan data
// lama), ditambah 3 kolom baru di akhir: Sub Kategori, Satuan, Cup Setara.
// Dibuat kalau belum ada; kalau sudah ada tapi belum punya kolom baru,
// di-upgrade dengan mengisi nilai wajar untuk baris lama (tanpa menghapus
// data histori).
// ---------------------------------------------------------------------
function setupLogSheet(ss) {
  let sheet = ss.getSheetByName(SHEET_LOG);

  if (!sheet) {
    sheet = ss.insertSheet(SHEET_LOG);
    sheet.getRange(1, 1, 1, 14).setValues([[
      "Waktu Input", "Tanggal", "Nama Rider", "Menu", "Kategori", "Harga",
      "Stock Awal", "Terjual", "Sisa", "Total Harga",
      "Sub Kategori", "Satuan", "Cup Setara", "ID Submission"
    ]]).setFontWeight("bold").setBackground("#1F6F50").setFontColor("#FFFFFF");
    sheet.setFrozenRows(1);
    sheet.autoResizeColumns(1, 14);
    return sheet;
  }

  const header = sheet.getRange(1, 1, 1, Math.max(sheet.getLastColumn(), 14)).getValues()[0];
  if (header[12] !== "Cup Setara") {
    sheet.getRange(1, 11, 1, 3).setValues([["Sub Kategori", "Satuan", "Cup Setara"]])
      .setFontWeight("bold").setBackground("#1F6F50").setFontColor("#FFFFFF");

    const lastRow = findLastDataRow(sheet, 1);
    if (lastRow > 1) {
      const oldRows = sheet.getRange(2, 1, lastRow - 1, 10).getValues();
      const fillData = oldRows.map(row => {
        const kategori = row[4];   // kolom E: Kategori
        const terjual = row[7];    // kolom H: Terjual
        const subKategori = kategori === "Cookies" ? "" : "Cup";
        const satuan = kategori === "Cookies" ? "pcs" : "cup";
        return [subKategori, satuan, terjual]; // data lama: konversi 1:1
      });
      sheet.getRange(2, 11, fillData.length, 3).setValues(fillData);
    }
  }
  if (header[13] !== "ID Submission") {
    sheet.getRange(1, 14).setValue("ID Submission")
      .setFontWeight("bold").setBackground("#1F6F50").setFontColor("#FFFFFF");
    // Baris lama tidak dapat ID (tidak bisa direvisi lewat Jeda Admin,
    // hanya baris baru sejak update ini yang bisa).
  }
  sheet.autoResizeColumns(1, 14);
  return sheet;
}

// ---------------------------------------------------------------------
// Sheet Rekap — selalu dibangun ulang (murni rumus, aman di-reset).
// ---------------------------------------------------------------------
function setupRekapSheet(ss, menuSheet, riderSheet) {
  let sheet = ss.getSheetByName(SHEET_REKAP);
  if (!sheet) sheet = ss.insertSheet(SHEET_REKAP);
  sheet.clear();
  // bersihkan validasi/data lama juga
  sheet.getDataRange().clearDataValidations();

  sheet.getRange("A1").setValue("Pilih Rider").setFontWeight("bold");
  sheet.getRange("A2").setValue("Pilih Tanggal").setFontWeight("bold");
  sheet.getRange("B2").setValue(new Date()).setNumberFormat("yyyy-mm-dd");

  const riderRule = SpreadsheetApp.newDataValidation()
    .requireValueInRange(riderSheet.getRange("A2:A200"), true)
    .setAllowInvalid(true)
    .build();
  sheet.getRange("B1").setDataValidation(riderRule);

  // header tabel: Menu | Kategori | Sub Kategori | Harga | Satuan |
  //               Stock Awal | Terjual | Sisa | Total Harga | Cup Setara
  const headerRow = 4;
  sheet.getRange(headerRow, 1, 1, 10).setValues([[
    "Menu", "Kategori", "Sub Kategori", "Harga", "Satuan",
    "Stock Awal", "Terjual", "Sisa", "Total Harga", "Cup Setara"
  ]]).setFontWeight("bold").setBackground("#EFEADF");

  const startRow = headerRow + 1;
  for (let i = 0; i < JUMLAH_SLOT_MENU; i++) {
    const r = startRow + i;
    const menuRow = i + 2; // baris di sheet Menu
    sheet.getRange(r, 1).setFormula(`=IFERROR(Menu!A${menuRow},"")`);
    sheet.getRange(r, 2).setFormula(`=IFERROR(Menu!B${menuRow},"")`);
    sheet.getRange(r, 3).setFormula(`=IFERROR(Menu!D${menuRow},"")`);
    sheet.getRange(r, 4).setFormula(`=IFERROR(Menu!C${menuRow},"")`);
    sheet.getRange(r, 5).setFormula(`=IFERROR(Menu!E${menuRow},"")`);
    sheet.getRange(r, 6).setFormula(
      `=IF($A${r}="","",SUMIFS(Log!$G:$G,Log!$C:$C,$B$1,Log!$B:$B,$B$2,Log!$D:$D,$A${r}))`);
    sheet.getRange(r, 7).setFormula(
      `=IF($A${r}="","",SUMIFS(Log!$H:$H,Log!$C:$C,$B$1,Log!$B:$B,$B$2,Log!$D:$D,$A${r}))`);
    sheet.getRange(r, 8).setFormula(`=IF($A${r}="","",F${r}-G${r})`);
    sheet.getRange(r, 9).setFormula(`=IF($A${r}="","",G${r}*D${r})`);
    sheet.getRange(r, 10).setFormula(
      `=IF($A${r}="","",SUMIFS(Log!$M:$M,Log!$C:$C,$B$1,Log!$B:$B,$B$2,Log!$D:$D,$A${r}))`);
  }
  const endRow = startRow + JUMLAH_SLOT_MENU - 1;

  const totMinRow = endRow + 1;
  const totCookRow = endRow + 2;
  const grandRow = endRow + 3;

  sheet.getRange(totMinRow, 1).setValue("Total Minuman").setFontWeight("bold");
  sheet.getRange(totCookRow, 1).setValue("Total Cookies").setFontWeight("bold");
  sheet.getRange(grandRow, 1).setValue("Grand Total").setFontWeight("bold").setBackground("#FFF2A8");

  ["F", "G", "H", "I", "J"].forEach((col) => {
    sheet.getRange(`${col}${totMinRow}`).setFormula(
      `=SUMIF($B${startRow}:$B${endRow},"Minuman",${col}${startRow}:${col}${endRow})`);
    sheet.getRange(`${col}${totCookRow}`).setFormula(
      `=SUMIF($B${startRow}:$B${endRow},"Cookies",${col}${startRow}:${col}${endRow})`);
    sheet.getRange(`${col}${grandRow}`).setFormula(
      `=${col}${totMinRow}+${col}${totCookRow}`).setBackground("#FFF2A8");
  });

  // Tabel Aturan Komisi — basis hitungan pakai kolom Cup Setara (J),
  // yang untuk Minuman = total cup (termasuk konversi botol), dan untuk
  // Cookies = total pcs (konversi selalu 1).
  const komisiHeaderRow = grandRow + 3;
  sheet.getRange(komisiHeaderRow, 1, 1, 3)
    .setValues([["Aturan Komisi", "Jumlah", "Komisi"]])
    .setFontWeight("bold").setBackground("#EFEADF");

  sheet.getRange(komisiHeaderRow + 1, 1).setValue("Minuman (cup)").setFontWeight("bold");
  sheet.getRange(komisiHeaderRow + 2, 1).setFormula(
    `="Tier 1: 1-"&Aturan!$B$2&" cup x Rp"&TEXT(Aturan!$C$2,"#,##0")`);
  sheet.getRange(komisiHeaderRow + 2, 2).setFormula(
    `=IF(J${totMinRow}>Aturan!$B$2,Aturan!$B$2,J${totMinRow})`);
  sheet.getRange(komisiHeaderRow + 2, 3).setFormula(`=B${komisiHeaderRow + 2}*Aturan!$C$2`);
  sheet.getRange(komisiHeaderRow + 3, 1).setFormula(
    `="Tier 2: "&(Aturan!$B$2+1)&"-200 cup x Rp"&TEXT(Aturan!$D$2,"#,##0")`);
  sheet.getRange(komisiHeaderRow + 3, 2).setFormula(
    `=IF(J${totMinRow}>Aturan!$B$2,J${totMinRow}-Aturan!$B$2,0)`);
  sheet.getRange(komisiHeaderRow + 3, 3).setFormula(`=B${komisiHeaderRow + 3}*Aturan!$D$2`);

  sheet.getRange(komisiHeaderRow + 4, 1).setValue("Cookies (pcs)").setFontWeight("bold");
  sheet.getRange(komisiHeaderRow + 5, 1).setFormula(
    `="Tier 1: 1-"&Aturan!$B$3&" pcs x Rp"&TEXT(Aturan!$C$3,"#,##0")`);
  sheet.getRange(komisiHeaderRow + 5, 2).setFormula(
    `=IF(J${totCookRow}>Aturan!$B$3,Aturan!$B$3,J${totCookRow})`);
  sheet.getRange(komisiHeaderRow + 5, 3).setFormula(`=B${komisiHeaderRow + 5}*Aturan!$C$3`);
  sheet.getRange(komisiHeaderRow + 6, 1).setFormula(
    `="Tier 2: "&(Aturan!$B$3+1)&"-200 pcs x Rp"&TEXT(Aturan!$D$3,"#,##0")`);
  sheet.getRange(komisiHeaderRow + 6, 2).setFormula(
    `=IF(J${totCookRow}>Aturan!$B$3,J${totCookRow}-Aturan!$B$3,0)`);
  sheet.getRange(komisiHeaderRow + 6, 3).setFormula(`=B${komisiHeaderRow + 6}*Aturan!$D$3`);

  const totalKomisiRow = komisiHeaderRow + 7;
  sheet.getRange(totalKomisiRow, 1).setValue("Total Komisi").setFontWeight("bold");
  sheet.getRange(totalKomisiRow, 3)
    .setFormula(`=C${komisiHeaderRow + 2}+C${komisiHeaderRow + 3}+C${komisiHeaderRow + 5}+C${komisiHeaderRow + 6}`)
    .setBackground("#FFF2A8").setFontWeight("bold");

  sheet.getRange(`D${startRow}:D${grandRow}`).setNumberFormat("Rp#,##0");
  sheet.getRange(`I${startRow}:I${grandRow}`).setNumberFormat("Rp#,##0");
  sheet.getRange(`C${komisiHeaderRow + 1}:C${totalKomisiRow}`).setNumberFormat("Rp#,##0");
  sheet.autoResizeColumns(1, 10);
  return sheet;
}

// ---------------------------------------------------------------------
// ID sesi harian = gabungan rider + tanggal (deterministik, tidak perlu
// random/UUID). Satu rider + satu tanggal = satu sesi/"submission" yang
// sama, baik diisi lewat tap +/- real-time maupun rekap sekali di akhir.
// ---------------------------------------------------------------------
function sessionIdOf(rider, tanggal) {
  return String(rider).trim() + "|" + String(tanggal).trim();
}

// Cari baris Log utk kombinasi rider+tanggal+menu tertentu. Return nomor
// baris (1-indexed di sheet) atau -1 kalau belum ada.
function findLogRow(logSheet, sessionId, menu) {
  const lastRow = findLastDataRow(logSheet, 1);
  if (lastRow <= 1) return -1;
  const ids = logSheet.getRange(2, 14, lastRow - 1, 1).getValues();
  const menus = logSheet.getRange(2, 4, lastRow - 1, 1).getValues();
  for (let i = 0; i < ids.length; i++) {
    if (ids[i][0] === sessionId && menus[i][0] === menu) return i + 2;
  }
  return -1;
}


// =====================================================================
// 2. doGet — dipanggil app untuk memuat daftar menu, rider, & aturan;
//    juga dipanggil utk mode real-time (getSession) & Jeda Admin
// =====================================================================
function doGet(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const action = e && e.parameter && e.parameter.action;

  if (action === "listSubmissions") return handleListSubmissions(ss, e.parameter);
  if (action === "getSubmission") return handleGetSubmission(ss, e.parameter.id);
  if (action === "getSession") return handleGetSession(ss, e.parameter.rider, e.parameter.tanggal);

  const menuSheet = ss.getSheetByName(SHEET_MENU);
  const riderSheet = ss.getSheetByName(SHEET_RIDER);
  const aturanSheet = ss.getSheetByName(SHEET_ATURAN);

  const menuData = menuSheet.getDataRange().getValues().slice(1)
    .filter(row => row[0])
    .map(row => ({
      menu: row[0], kategori: row[1], harga: row[2],
      subKategori: row[3], satuan: row[4], konversi: row[5] || 1
    }));

  const riders = riderSheet.getDataRange().getValues().slice(1)
    .map(row => row[0]).filter(String);

  const aturanRows = aturanSheet.getDataRange().getValues().slice(1, 3); // baris 2-3 saja (Minuman, Cookies)
  const aturan = {};
  aturanRows.forEach(row => {
    if (!row[0]) return;
    aturan[row[0]] = { batas: row[1], tarif1: row[2], tarif2: row[3] };
  });
  const operasional = aturanSheet.getRange("B5").getValue() || 0;

  return ContentService
    .createTextOutput(JSON.stringify({
      status: "ok", menu: menuData, riders: riders, aturan: aturan, operasional: operasional
    }))
    .setMimeType(ContentService.MimeType.JSON);
}

// ---------------------------------------------------------------------
// Untuk Jeda Admin: daftar semua submission (1 baris Setoran = 1 submission),
// terbaru dulu. Bisa difilter lewat query ?rider=...&tanggal=YYYY-MM-DD
// ---------------------------------------------------------------------
function handleListSubmissions(ss, params) {
  const setoranSheet = ss.getSheetByName(SHEET_SETORAN);
  const lastRow = findLastDataRow(setoranSheet, 1);
  let list = [];
  if (lastRow > 1) {
    const rows = setoranSheet.getRange(2, 1, lastRow - 1, 10).getValues();
    list = rows
      .filter(row => row[9]) // hanya yang punya ID Submission (baris lama tanpa ID dilewati)
      .map(row => ({
        id: row[9], waktuInput: row[0], tanggal: Utilities.formatDate(new Date(row[1]), Session.getScriptTimeZone(), "yyyy-MM-dd"),
        rider: row[2], totalPenjualan: row[3], totalKomisi: row[4],
        modal: row[5], qris: row[6], operasional: row[7], setoranCash: row[8]
      }));
  }
  if (params) {
    if (params.rider) list = list.filter(s => s.rider === params.rider);
    if (params.tanggal) list = list.filter(s => s.tanggal === params.tanggal);
  }
  list.reverse(); // terbaru dulu
  return ContentService.createTextOutput(JSON.stringify({ status: "ok", submissions: list }))
    .setMimeType(ContentService.MimeType.JSON);
}

// ---------------------------------------------------------------------
// Untuk Jeda Admin: detail item-item 1 submission (dari sheet Log),
// dicocokkan lewat ID Submission.
// ---------------------------------------------------------------------
function handleGetSubmission(ss, id) {
  if (!id) return jsonError("ID submission tidak ada");
  const logSheet = ss.getSheetByName(SHEET_LOG);
  const setoranSheet = ss.getSheetByName(SHEET_SETORAN);
  const menuSheet = ss.getSheetByName(SHEET_MENU);
  const menuLookup = {};
  menuSheet.getDataRange().getValues().slice(1).forEach(row => {
    if (row[0]) menuLookup[row[0]] = { satuan: row[4], konversi: row[5] || 1 };
  });

  const lastRow = findLastDataRow(logSheet, 1);
  const items = [];
  let rider = "", tanggal = "";
  if (lastRow > 1) {
    const rows = logSheet.getRange(2, 1, lastRow - 1, 14).getValues();
    rows.forEach(row => {
      if (row[13] !== id) return;
      rider = row[2]; tanggal = row[1];
      const info = menuLookup[row[3]] || { satuan: row[11] || "cup", konversi: 1 };
      items.push({
        menu: row[3], kategori: row[4], harga: row[5],
        stockAwal: row[6], terjual: row[7],
        subKategori: row[10], satuan: info.satuan, konversi: info.konversi
      });
    });
  }
  if (!items.length) return jsonError("Submission tidak ditemukan (mungkin data lama sebelum fitur ini ada)");

  let modal = 0, qris = 0;
  const lastSetoranRow = findLastDataRow(setoranSheet, 1);
  if (lastSetoranRow > 1) {
    const setoranRows = setoranSheet.getRange(2, 1, lastSetoranRow - 1, 10).getValues();
    const match = setoranRows.find(row => row[9] === id);
    if (match) { modal = match[5] || 0; qris = match[6] || 0; }
  }

  const tanggalStr = tanggal instanceof Date
    ? Utilities.formatDate(tanggal, Session.getScriptTimeZone(), "yyyy-MM-dd") : tanggal;
  return ContentService
    .createTextOutput(JSON.stringify({ status: "ok", id, rider, tanggal: tanggalStr, items, modal, qris }))
    .setMimeType(ContentService.MimeType.JSON);
}

// ---------------------------------------------------------------------
// Untuk mode real-time Jeda Rider: ambil status Stock Awal/Terjual yang
// sudah tersimpan sejauh ini utk rider+tanggal tertentu (dipanggil saat
// app dibuka/dibuka ulang, supaya angka yang sudah di-tap tidak hilang).
// Kalau belum ada sesi sama sekali, items dikembalikan kosong (wajar,
// berarti belum ada tap/isian apa pun hari itu).
// ---------------------------------------------------------------------
function handleGetSession(ss, rider, tanggal) {
  if (!rider || !tanggal) return jsonError("rider/tanggal tidak ada");
  const id = sessionIdOf(rider, tanggal);
  const logSheet = ss.getSheetByName(SHEET_LOG);
  const setoranSheet = ss.getSheetByName(SHEET_SETORAN);

  const items = {}; // keyed by nama menu -> {stockAwal, terjual}
  const lastRow = findLastDataRow(logSheet, 1);
  if (lastRow > 1) {
    const rows = logSheet.getRange(2, 1, lastRow - 1, 14).getValues();
    rows.forEach(row => {
      if (row[13] !== id) return;
      items[row[3]] = { stockAwal: row[6], terjual: row[7] };
    });
  }

  let modal = 0, qris = 0;
  const lastSetoranRow = findLastDataRow(setoranSheet, 1);
  if (lastSetoranRow > 1) {
    const setoranRows = setoranSheet.getRange(2, 1, lastSetoranRow - 1, 10).getValues();
    const match = setoranRows.find(row => row[9] === id);
    if (match) { modal = match[5] || 0; qris = match[6] || 0; }
  }

  return ContentService
    .createTextOutput(JSON.stringify({ status: "ok", id, items, modal, qris }))
    .setMimeType(ContentService.MimeType.JSON);
}

// =====================================================================
// 3. doPost — entry data penjualan & kelola menu
// =====================================================================
function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const ss = SpreadsheetApp.getActiveSpreadsheet();

    // -------------------------------------------------------------
    // Mode real-time: tap +/- (adjustItem) atau isi angka langsung
    // (setItemField). Upsert baris Log per (rider, tanggal, menu).
    // -------------------------------------------------------------
    if (data.action === "adjustItem" || data.action === "setItemField") {
      const lock = LockService.getScriptLock();
      lock.waitLock(10000);
      try {
        const logSheet = ss.getSheetByName(SHEET_LOG);
        const id = sessionIdOf(data.rider, data.tanggal);
        let rowIndex = findLogRow(logSheet, id, data.menu);
        const konversi = data.konversi || 1;

        let stockAwal = 0, terjual = 0;
        if (rowIndex > 0) {
          const existing = logSheet.getRange(rowIndex, 7, 1, 2).getValues()[0];
          stockAwal = existing[0] || 0;
          terjual = existing[1] || 0;
        }

        if (data.action === "adjustItem") {
          if (data.field === "terjual") terjual = Math.max(0, terjual + data.delta);
          else stockAwal = Math.max(0, stockAwal + data.delta);
        } else { // setItemField -> nilai absolut
          if (data.field === "terjual") terjual = Math.max(0, data.value);
          else stockAwal = Math.max(0, data.value);
        }

        const rowValues = [
          new Date(), data.tanggal, data.rider, data.menu, data.kategori, data.harga,
          stockAwal, terjual, stockAwal - terjual, terjual * data.harga,
          data.subKategori || "", data.satuan || "", terjual * konversi, id
        ];
        if (rowIndex > 0) {
          logSheet.getRange(rowIndex, 1, 1, 14).setValues([rowValues]);
        } else {
          const appendRow = findLastDataRow(logSheet, 1) + 1;
          logSheet.getRange(appendRow, 1, 1, 14).setValues([rowValues]);
        }
        return jsonOkData({ stockAwal, terjual });
      } finally {
        lock.releaseLock();
      }
    }

    // -------------------------------------------------------------
    // Tutup sesi hari ini: simpan/timpa ringkasan Setoran (Modal, QRIS,
    // Komisi, Operasional, Setoran Cash) utk rider+tanggal ybs. Data
    // per-menu TIDAK dikirim di sini karena sudah tersimpan real-time
    // lewat adjustItem/setItemField sepanjang hari.
    // -------------------------------------------------------------
    if (data.action === "closeSession") {
      const lock = LockService.getScriptLock();
      lock.waitLock(10000);
      try {
        const id = sessionIdOf(data.rider, data.tanggal);
        const setoranSheet = ss.getSheetByName(SHEET_SETORAN);
        const now = new Date();
        const rowValues = [
          now, data.tanggal, data.rider,
          data.totalPenjualan || 0, data.totalKomisi || 0,
          data.modal || 0, data.qris || 0, data.operasional || 0,
          data.setoranCash || 0, id
        ];
        const lastRow = findLastDataRow(setoranSheet, 1);
        let rowIndex = -1;
        if (lastRow > 1) {
          const ids = setoranSheet.getRange(2, 10, lastRow - 1, 1).getValues();
          for (let i = 0; i < ids.length; i++) {
            if (ids[i][0] === id) { rowIndex = i + 2; break; }
          }
        }
        if (rowIndex > 0) {
          setoranSheet.getRange(rowIndex, 1, 1, 10).setValues([rowValues]);
        } else {
          const appendRow = findLastDataRow(setoranSheet, 1) + 1;
          setoranSheet.getRange(appendRow, 1, 1, 10).setValues([rowValues]);
        }
        return jsonOk();
      } finally {
        lock.releaseLock();
      }
    }

    if (data.action === "entry") {
      const lock = LockService.getScriptLock();
      lock.waitLock(10000);
      try {
        const submissionId = Utilities.getUuid();
        const logSheet = ss.getSheetByName(SHEET_LOG);
        const now = new Date();
        const rows = data.items.map(item => {
          const konversi = item.konversi || 1;
          return [
            now, data.tanggal, data.rider, item.menu, item.kategori, item.harga,
            item.stockAwal, item.terjual, item.stockAwal - item.terjual,
            item.terjual * item.harga,
            item.subKategori || "", item.satuan || "", item.terjual * konversi,
            submissionId
          ];
        });
        if (rows.length) {
          const appendRow = findLastDataRow(logSheet, 1) + 1;
          logSheet.getRange(appendRow, 1, rows.length, 14).setValues(rows);
        }

        // Simpan juga ringkasan setoran (1 baris per pengiriman)
        const setoranSheet = ss.getSheetByName(SHEET_SETORAN);
        const setoranRow = [
          now, data.tanggal, data.rider,
          data.totalPenjualan || 0, data.totalKomisi || 0,
          data.modal || 0, data.qris || 0, data.operasional || 0,
          data.setoranCash || 0, submissionId
        ];
        const setoranAppendRow = findLastDataRow(setoranSheet, 1) + 1;
        setoranSheet.getRange(setoranAppendRow, 1, 1, 10).setValues([setoranRow]);

        return jsonOk();
      } finally {
        lock.releaseLock();
      }
    }

    if (data.action === "addMenu" || data.action === "updateMenu") {
      const lock = LockService.getScriptLock();
      lock.waitLock(10000); // tunggu maks 10 detik kalau ada request lain yg sedang jalan
      try {
        const menuSheet = ss.getSheetByName(SHEET_MENU);
        const values = menuSheet.getDataRange().getValues();
        const key = data.action === "updateMenu" ? data.oldMenu : data.menu;
        let rowIndex = -1;
        for (let i = 1; i < values.length; i++) {
          if (values[i][0] === key) { rowIndex = i + 1; break; }
        }

        // Kalau TAMBAH menu baru tapi namanya sudah ada -> tolak, jangan menimpa.
        if (data.action === "addMenu" && rowIndex > 0) {
          return jsonError(
            `Menu dengan nama "${data.menu}" sudah ada. Gunakan tombol Edit ` +
            `kalau mau mengubahnya, atau pakai nama lain yang unik (misal ` +
            `"${data.menu} 1 Liter").`
          );
        }
        // Kalau EDIT tapi nama barunya bentrok dengan menu lain -> tolak juga.
        if (data.action === "updateMenu" && data.menu !== data.oldMenu) {
          const clashIndex = values.findIndex((row, i) => i > 0 && row[0] === data.menu);
          if (clashIndex > 0) {
            return jsonError(`Nama "${data.menu}" sudah dipakai menu lain. Pilih nama lain.`);
          }
        }

        const subKategori = data.subKategori || "";
        const sk = deriveSatuanKonversi(data.kategori, subKategori);
        const rowValues = [
          data.menu, data.kategori, data.harga, subKategori, sk.satuan, sk.konversi
        ];
        if (rowIndex > 0) {
          menuSheet.getRange(rowIndex, 1, 1, 6).setValues([rowValues]);
        } else {
          const appendRow = findLastDataRow(menuSheet, 1) + 1;
          menuSheet.getRange(appendRow, 1, 1, 6).setValues([rowValues]);
        }
        return jsonOk();
      } finally {
        lock.releaseLock();
      }
    }

    if (data.action === "deleteMenu") {
      const lock = LockService.getScriptLock();
      lock.waitLock(10000);
      try {
        const menuSheet = ss.getSheetByName(SHEET_MENU);
        const values = menuSheet.getDataRange().getValues();
        for (let i = 1; i < values.length; i++) {
          if (values[i][0] === data.menu) {
            menuSheet.deleteRow(i + 1);
            break;
          }
        }
        return jsonOk();
      } finally {
        lock.releaseLock();
      }
    }

    if (data.action === "verifyPin") {
      const ok = checkPin(ss, data.pin);
      return ok ? jsonOk() : jsonError("PIN salah");
    }

    if (data.action === "updateSubmission") {
      if (!checkPin(ss, data.pin)) return jsonError("PIN salah");
      const lock = LockService.getScriptLock();
      lock.waitLock(10000);
      try {
        const logSheet = ss.getSheetByName(SHEET_LOG);
        const setoranSheet = ss.getSheetByName(SHEET_SETORAN);
        const id = data.submissionId;

        // Hapus baris Log lama utk submission ini (dari bawah ke atas)
        const lastLogRow = findLastDataRow(logSheet, 1);
        if (lastLogRow > 1) {
          const idCol = logSheet.getRange(2, 14, lastLogRow - 1, 1).getValues();
          for (let i = idCol.length - 1; i >= 0; i--) {
            if (idCol[i][0] === id) logSheet.deleteRow(i + 2);
          }
        }

        // Tulis ulang baris Log baru (hasil koreksi) dengan ID yang sama
        const now = new Date();
        const rows = data.items.map(item => {
          const konversi = item.konversi || 1;
          return [
            now, data.tanggal, data.rider, item.menu, item.kategori, item.harga,
            item.stockAwal, item.terjual, item.stockAwal - item.terjual,
            item.terjual * item.harga,
            item.subKategori || "", item.satuan || "", item.terjual * konversi,
            id
          ];
        });
        if (rows.length) {
          const appendRow = findLastDataRow(logSheet, 1) + 1;
          logSheet.getRange(appendRow, 1, rows.length, 14).setValues(rows);
        }

        // Update baris Setoran yang sesuai (cari lewat ID, timpa di tempat)
        const lastSetoranRow = findLastDataRow(setoranSheet, 1);
        if (lastSetoranRow > 1) {
          const idCol = setoranSheet.getRange(2, 10, lastSetoranRow - 1, 1).getValues();
          for (let i = 0; i < idCol.length; i++) {
            if (idCol[i][0] === id) {
              setoranSheet.getRange(i + 2, 1, 1, 10).setValues([[
                now, data.tanggal, data.rider,
                data.totalPenjualan || 0, data.totalKomisi || 0,
                data.modal || 0, data.qris || 0, data.operasional || 0,
                data.setoranCash || 0, id
              ]]);
              break;
            }
          }
        }
        return jsonOk();
      } finally {
        lock.releaseLock();
      }
    }

    if (data.action === "deleteSubmission") {
      if (!checkPin(ss, data.pin)) return jsonError("PIN salah");
      const lock = LockService.getScriptLock();
      lock.waitLock(10000);
      try {
        const logSheet = ss.getSheetByName(SHEET_LOG);
        const setoranSheet = ss.getSheetByName(SHEET_SETORAN);
        const id = data.submissionId;

        const lastLogRow = findLastDataRow(logSheet, 1);
        if (lastLogRow > 1) {
          const idCol = logSheet.getRange(2, 14, lastLogRow - 1, 1).getValues();
          for (let i = idCol.length - 1; i >= 0; i--) {
            if (idCol[i][0] === id) logSheet.deleteRow(i + 2);
          }
        }
        const lastSetoranRow = findLastDataRow(setoranSheet, 1);
        if (lastSetoranRow > 1) {
          const idCol = setoranSheet.getRange(2, 10, lastSetoranRow - 1, 1).getValues();
          for (let i = idCol.length - 1; i >= 0; i--) {
            if (idCol[i][0] === id) setoranSheet.deleteRow(i + 2);
          }
        }
        return jsonOk();
      } finally {
        lock.releaseLock();
      }
    }

    return jsonError("Aksi tidak dikenali: " + data.action);

  } catch (err) {
    return jsonError(err.message);
  }
}

function checkPin(ss, pin) {
  const aturanSheet = ss.getSheetByName(SHEET_ATURAN);
  const correctPin = String(aturanSheet.getRange("B9").getValue()).trim();
  return String(pin || "").trim() === correctPin;
}

function jsonOk() {
  return ContentService.createTextOutput(JSON.stringify({ status: "ok" }))
    .setMimeType(ContentService.MimeType.JSON);
}
function jsonOkData(extra) {
  return ContentService.createTextOutput(JSON.stringify(Object.assign({ status: "ok" }, extra)))
    .setMimeType(ContentService.MimeType.JSON);
}
function jsonError(message) {
  return ContentService.createTextOutput(JSON.stringify({ status: "error", message: message }))
    .setMimeType(ContentService.MimeType.JSON);
}
